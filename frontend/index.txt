    <script>
      (function () {
        var year = new Date().getFullYear();
        document.getElementById("year").textContent = year;
        document.getElementById("footYear").textContent = year;

        // Riporta i recapiti da company-info.json, se raggiungibile.
        fetch("company-info.json", { cache: "no-store" })
          .then(function (r) {
            if (!r.ok) throw new Error("company-info non disponibile");
            return r.json();
          })
          .then(function (data) {
            var c = (data && data.company) || {};
            var shown = false;

            if (c.phone) {
              var pl = document.getElementById("phoneLink");
              pl.textContent = c.phone;
              pl.href = "tel:" + String(c.phone).replace(/\s+/g, "");
              document.getElementById("contactPhone").hidden = false;
              shown = true;
            }

            if (c.email) {
              var el = document.getElementById("emailLink");
              el.textContent = c.email;
              el.href = "mailto:" + c.email;
              document.getElementById("contactEmail").hidden = false;
              shown = true;
            }

            if (shown) {
              document.getElementById("contacts").classList.add("is-visible");
            }
          })
          .catch(function () {
            /* Nessun recapito da mostrare: il blocco resta nascosto. */
          });
      })();
    </script>
